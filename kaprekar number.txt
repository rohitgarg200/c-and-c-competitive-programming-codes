void kaprekarNumbers(int p, int q) {
for(int i=p;i<=q;i++){
    int s=pow(i,2);
    int r=i;
    int count=0;;
    while(r!=0){
        r=r/10;
        count++;

    }
    int right,left;
    left=s/pow(10,count);
    int z=pow(10,count);
    right=s%z;
    if(left+right==i){
        printf("%d ",i);
    }

}

}
