#include <bits/stdc++.h>

using namespace std;

// Complete the pangrams function below.
string pangrams(string s) {
    int a[26]={0};
    transform(s.begin(), s.end(), s.begin(), ::tolower);
    for(int i=0;i<s.size();i++){
        if(s[i]>='a'&& s[i]<='z'){
            a[s[i]-'a']++;
        }
    }
    for(int i=0;i<26;i++){
        if(a[i]==0){
        return "not pangram";
        
        }
    }
return "pangram";

}

int main()
{
    ofstream fout(getenv("OUTPUT_PATH"));

    string s;
    getline(cin, s);

    string result = pangrams(s);

    fout << result << "\n";

    fout.close();

    return 0;
}
